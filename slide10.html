API Integration Test Example (JavaScript)
Using Supertest with Express
const request = require('supertest'); const app = require('../app'); const db = require('../db'); describe('User API Integration Tests', () => { // Setup: Connect to test database beforeAll(async () => { await db.connect('mongodb://localhost:27017/test_db'); }); // Clean up after tests afterAll(async () => { await db.disconnect(); }); // Reset database between tests afterEach(async () => { await db.clearCollections(); }); test('GET /api/users should return all users', async () => { // Arrange: Create test data await db.users.insertMany([ { name: 'Alice', email: 'alice@example.com' }, { name: 'Bob', email: 'bob@example.com' } ]); // Act: Make API request const response = await request(app) .get('/api/users') .expect('Content-Type', /json/) .expect(200); // Assert: Check response body expect(response.body.length).toBe(2); expect(response.body[0].name).toBe('Alice'); expect(response.body[1].name).toBe('Bob'); }); test('POST /api/users should create a new user', async () => { // Act: Make API request to create user const response = await request(app) .post('/api/users') .send({ name: 'Charlie', email: 'charlie@example.com' }) .expect('Content-Type', /json/) .expect(201); // Assert: Verify response and database state expect(response.body.name).toBe('Charlie'); // Verify user was actually saved to database const savedUser = await db.users.findOne({ email: 'charlie@example.com' }); expect(savedUser).toBeTruthy(); expect(savedUser.name).toBe('Charlie'); }); });

Best Practice: Use Test Containers

For database integration tests, consider using Testcontainers to spin up isolated, real database instances in Docker containers during testing, providing the most realistic environment without affecting production data.
